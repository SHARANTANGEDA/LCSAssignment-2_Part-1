16 InfixPostfix::InfixPostfix(int len, char arr[])
17 {
18  stack<char> s; //<creating a character stack 's'
19  stringstream post; //<creating a stringstream called post
20  //iterating through the character array 'arr[]'
21  for (int i = 0; i < len; i++) {
22  /*if 'i'th element of 'arr[]' is not an operator symbol
23  then append 'arr[i]' to post*/
24  if ((arr[i] != '~') && (arr[i] != 'V') && (arr[i] != '^') && (arr[i] != '>') && (arr[i] != ')') && (arr[i] != '(')) {
25  post << arr[i];
26  }
27  /*if stack 's' is empty and 'i'th element of 'arr[]' is an operator
28  push 'arr[i]' onto top of stack s */
29  if ((s.empty()) && ((arr[i] == '~') || (arr[i] == 'V') || (arr[i] == '^') || (arr[i] == '(') || (arr[i] == '>') || (arr[i] == ')'))) {
30  s.push(arr[i]);
31  }
32  //if stack 's' is not empty and 'i'th element of 'arr[]' is an operator we proceed based on type of operator
33  if(!(s.empty()) && ((arr[i] == '~') || (arr[i] == 'V') || (arr[i] == '^') || (arr[i] == '(') || (arr[i] == '>') || (arr[i] == ')'))) {
34  /* if 'arr[i]' is '('
35  push 'arr[i]' onto top of stack s*/
36  if(arr[i]=='(') {
37  s.push(arr[i]);
38  }
39  /*else if 'arr[i]' is ')'
40  */
41  else if(arr[i]==')') {
42  /*traverse down the stack s
43  * by appending top most element to 'post' and then 'pop'ing stack
44  * until we reach a '('
45  */
46  while (s.top()!='(') {
47  post<<s.top();
48  s.pop();
49  }

51  s.pop();
52  }
53  //if arr[i] is any other operator
54  else {
55  /*we traverse down the stack s

57  * until precedence of top of stack is less than precedence of 'arr[i]' (function for precedence is is defined later )
58  */
59  while (precedence(s.top())>=precedence(arr[i])) {

70  * until stack is empty
